<?php
      $tax_rate = 0.2;
      
      function calculate_running_total($price, $quantity)
      {
          global $tax_rate;
          static $running_total = 0;
          $total = $price * $quantity;
          $tax   = $total * $tax_rate;
          $running_total = $running_total + $total + $tax;
          return $running_total;
      }
      ?>

      <head>
        <title>Global and Static Variables</title>
      </head>

          </tr>
