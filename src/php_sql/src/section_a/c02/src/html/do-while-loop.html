<?php
      $packs = 5;
      $price = 1.99;
      ?>
    

    

       <html>
         <head>
           <title>do while Loop</title>
         </head>
         <body>
           <h1>The Candy Store</h1>
           <h2>Prices for Multiple Packs</h2>
           <p>
             do {
             echo $packs;
             echo " packs cost $";
             echo $price * $packs;
             echo "<br />";
             $packs--;
             } while ($packs > 0 );
             ?>
           </p>
         </body>
       </html>
